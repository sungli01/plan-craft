interface TimeInfo {
  totalMinutes: number;    // ì´ ì˜ˆìƒ ì‹œê°„ (29ë¶„)
  elapsedMinutes: number;  // ê²½ê³¼ ì‹œê°„
  remainingMinutes: number; // ë‚¨ì€ ì‹œê°„
  progressPercent: number; // ì§„í–‰ë¥  (%)
}

const PHASE_DURATIONS = {
  'G1_CORE_LOGIC': 3,
  'G2_DATABASE': 4,
  'G3_API': 5,
  'G4_AUTH': 3,
  'G5_FRONTEND': 6,
  'G6_TESTING': 3,
  'G7_OPTIMIZATION': 2,
  'G8_SECURITY': 1,
  'G9_DEPLOYMENT': 1,
  'G10_HANDOVER': 1,
};

function calculateTimeInfo(project: Project): TimeInfo {
  const totalMinutes = Object.values(PHASE_DURATIONS).reduce((a, b) => a + b, 0);
  
  // í˜„ì¬ê¹Œì§€ ì™„ë£Œëœ ë‹¨ê³„ì˜ ì‹œê°„ í•©ì‚°
  let elapsedMinutes = 0;
  for (const phase of project.completedPhases) {

  }
  
  // í˜„ì¬ ì§„í–‰ ì¤‘ì¸ ë‹¨ê³„ì˜ ê²½ê³¼ ì‹œê°„ ì¶”ê°€
  if (project.currentPhase) {
    const phaseStartTime = project.phaseStartTime;
    const now = Date.now();
    const currentPhaseElapsed = Math.floor((now - phaseStartTime) / 60000);
    elapsedMinutes += Math.min(
      currentPhaseElapsed,
      PHASE_DURATIONS[project.currentPhase]
    );
  }
  
  const remainingMinutes = Math.max(0, totalMinutes - elapsedMinutes);
  const progressPercent = Math.floor((elapsedMinutes / totalMinutes) * 100);
  
  return {
    totalMinutes,
    elapsedMinutes,
    remainingMinutes,
    progressPercent,
  };
}
